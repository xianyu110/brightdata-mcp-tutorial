💻 代码示例

基础搜索示例

from brightdata import bdclient

# 初始化客户端
client = bdclient(api_token="your_api_token_here")

# 执行搜索
results = client.search("Python web scraping tutorial")

# 保存结果
with open("results.html", "w") as f:
    f.write(results)
